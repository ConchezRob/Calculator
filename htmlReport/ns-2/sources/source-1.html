     public Set<Integer> ensembleChiffre(int pNombre) {
         Set<Integer> chiffres = new HashSet<>();
         int n = Math.abs(pNombre);
         if (n == 0) {
             chiffres.add(0);
         }
         while (n > 0) {
             chiffres.add(n % 10);
             n /= 10;
         }
         return chiffres;
